# Text-to-Speech Web UI Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a Text-to-Speech section to the Lightning Whisper MLX web UI, with a FastAPI backend endpoint and a new frontend section that lets users type text, configure TTS parameters, generate speech, and play/download the resulting WAV file.

**Architecture:** Backend gains a `POST /api/tts` endpoint that accepts text + TTS parameters, runs `LightningTTSMLX.generate()` in a background thread (same job pattern as transcription), and serves the resulting WAV via `GET /api/tts-jobs/{job_id}/audio`. Frontend adds a Section 5 "Text to Speech" with a textarea, parameter controls (steps, speed), a generate button, and an inline audio player with download.

**Tech Stack:** Python (FastAPI, existing `LightningTTSMLX`), Next.js (React), existing brutalist UI component library.

```python
def _run_tts(job_id: str, output_path: str, text: str,
             steps: int, speed: float, seed: int | None) -> None:
    """Run TTS generation in a background thread."""
    try:
        with _jobs_lock:
            _jobs[job_id]["status"] = JobStatus.processing

        from .tts import LightningTTSMLX
        tts = LightningTTSMLX()
        tts.generate(text=text, output_path=output_path, steps=steps, speed=speed, seed=seed)

        with _jobs_lock:
            _jobs[job_id]["status"] = JobStatus.completed
            _jobs[job_id]["result"] = {"audio_path": output_path}
    except Exception as e:
        with _jobs_lock:
            _jobs[job_id]["status"] = JobStatus.failed
            _jobs[job_id]["error"] = str(e)
        # Clean up on failure
        Path(output_path).unlink(missing_ok=True)
```

**Step 2: Add the POST /api/tts endpoint**

@app.post("/api/tts", status_code=202)
async def text_to_speech(
    text: str = Form(),
    steps: int = Form(default=8),
    speed: float = Form(default=1.0),
    seed: int | None = Form(default=None),
) -> dict[str, str]:
    """Submit text for speech synthesis."""
    if not text.strip():
        raise HTTPException(status_code=422, detail="text must not be empty")

    job_id = str(uuid.uuid4())
    output_path = str(Path(tempfile.gettempdir()) / f"tts_{job_id}.wav")

    with _jobs_lock:
        _jobs[job_id] = {
            "status": JobStatus.queued,
            "result": None,
            "error": None,
        }

    thread = threading.Thread(
        target=_run_tts,
        args=(job_id, output_path, text, steps, speed, seed),
        daemon=True,
    )
    thread.start()

    return {"job_id": job_id, "status": "queued"}
```

from fastapi.responses import FileResponse

@app.get("/api/tts-jobs/{job_id}/audio")
def get_tts_audio(job_id: str) -> FileResponse:
    """Download the generated audio file."""
    with _jobs_lock:
        if job_id not in _jobs:
            raise HTTPException(status_code=404, detail="Job not found")
        job = _jobs[job_id]

    result = job.get("result")
    if not result or not result.get("audio_path"):
        raise HTTPException(status_code=404, detail="Audio not available")

    audio_path = Path(result["audio_path"])
    if not audio_path.exists():
        raise HTTPException(status_code=404, detail="Audio file not found")

    return FileResponse(audio_path, media_type="audio/wav", filename=f"tts_{job_id}.wav")
```

export interface TtsParams {
  text: string
  steps: number
  speed: number
}

export async function startTts(params: TtsParams): Promise<{ job_id: string; status: string }> {
  const form = new FormData()
  form.append("text", params.text)
  form.append("steps", String(params.steps))
  form.append("speed", String(params.speed))

  const res = await fetch(`${API_BASE}/api/tts`, { method: "POST", body: form })
  if (!res.ok) {
    const err = await res.json().catch(() => ({ detail: res.statusText }))
    throw new Error(err.detail ?? `Failed to start TTS: ${res.status}`)
  }
  return res.json()
}

export function getTtsAudioUrl(jobId: string): string {
  return `${API_BASE}/api/tts-jobs/${jobId}/audio`
}

### Task 5: Build the TTS section component

**Files:**
- Create: `lightning-whisper-mlx-ui/components/sections/text-to-speech.tsx`

**Step 1: Create the component**

```tsx
"use client"

import { useState, useRef } from "react"
import { Loader2, Volume2, Download } from "lucide-react"
import { Button } from "@/components/ui/button"
import { Label } from "@/components/ui/label"
import { Badge } from "@/components/ui/badge"
import { startTts, pollJob, getTtsAudioUrl } from "@/lib/api"

interface TextToSpeechProps {
  status: "idle" | "queued" | "processing" | "completed" | "failed"
  onStatusChange: (status: "idle" | "queued" | "processing" | "completed" | "failed") => void
  audioJobId: string | null
  onAudioJobIdChange: (id: string | null) => void
  error: string | null
  onErrorChange: (error: string | null) => void
}

const statusVariant: Record<string, "ghost" | "warning" | "success" | "error"> = {
  idle: "ghost",
  queued: "warning",
  processing: "warning",
  completed: "success",
  failed: "error",
}

export function TextToSpeech({
  status,
  onStatusChange,
  audioJobId,
  onAudioJobIdChange,
  error,
  onErrorChange,
}: TextToSpeechProps) {
  const [text, setText] = useState("")
  const [steps, setSteps] = useState(8)
  const [speed, setSpeed] = useState(1.0)
  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null)

  const isRunning = status === "queued" || status === "processing"

  const handleGenerate = async () => {
    if (!text.trim()) return

    onStatusChange("queued")
    onErrorChange(null)
    onAudioJobIdChange(null)

    try {
      const { job_id } = await startTts({ text, steps, speed })

      pollRef.current = setInterval(async () => {
        try {
          const job = await pollJob(job_id)
          onStatusChange(job.status)

          if (job.status === "completed") {
            onAudioJobIdChange(job_id)
            if (pollRef.current) clearInterval(pollRef.current)
          } else if (job.status === "failed") {
            onErrorChange(job.error ?? "TTS generation failed")
            if (pollRef.current) clearInterval(pollRef.current)
          }
        } catch {
          onErrorChange("Lost connection to server")
          if (pollRef.current) clearInterval(pollRef.current)
        }
      }, 1000)
    } catch (e) {
      onStatusChange("failed")
      onErrorChange(e instanceof Error ? e.message : "Failed to start TTS")
    }
  }

  return (
    <div className="space-y-4">
      {/* Text input */}
      <div className="space-y-2">
        <Label htmlFor="tts-text">Text</Label>
        <textarea
          id="tts-text"
          value={text}
          onChange={(e) => setText(e.target.value)}
          placeholder="Enter text to synthesize..."
          rows={4}
          className="w-full rounded-sm border-2 border-border bg-background px-3 py-2 text-sm font-medium placeholder:text-muted-foreground focus:outline-none focus:ring-2 focus:ring-foreground/50 resize-y"
        />
        <p className="text-xs text-muted-foreground">{text.length} characters</p>
      </div>

      {/* Parameters */}
      <div className="grid grid-cols-2 gap-4">
        <div className="space-y-2">
          <Label htmlFor="tts-steps">Diffusion Steps</Label>
          <input
            id="tts-steps"
            type="number"
            min={1}
            max={64}
            value={steps}
            onChange={(e) => setSteps(Number(e.target.value))}
            className="w-full rounded-sm border-2 border-border bg-background px-3 py-2 text-sm font-medium focus:outline-none focus:ring-2 focus:ring-foreground/50"
          />
          <p className="text-xs text-muted-foreground">More = better quality, slower</p>
        </div>
        <div className="space-y-2">
          <Label htmlFor="tts-speed">Speed</Label>
          <input
            id="tts-speed"
            type="number"
            min={0.5}
            max={2.0}
            step={0.1}
            value={speed}
            onChange={(e) => setSpeed(Number(e.target.value))}
            className="w-full rounded-sm border-2 border-border bg-background px-3 py-2 text-sm font-medium focus:outline-none focus:ring-2 focus:ring-foreground/50"
          />
          <p className="text-xs text-muted-foreground">1.0 = normal speed</p>
        </div>
      </div>

      {/* Generate button */}
      <Button
        onClick={handleGenerate}
        disabled={!text.trim() || isRunning}
        className="w-full h-12 text-base"
      >
        {isRunning ? (
          <>
            <Loader2 className="h-5 w-5 animate-spin" />
            {status === "queued" ? "Queued..." : "Generating..."}
          </>
        ) : (
          <>
            <Volume2 className="h-5 w-5" />
            Generate Speech
          </>
        )}
      </Button>

      {/* Status badge */}
      {status !== "idle" && (
        <div className="flex items-center gap-2">
          <Badge variant={statusVariant[status]}>
            {status.charAt(0).toUpperCase() + status.slice(1)}
          </Badge>
        </div>
      )}

      {/* Error */}
      {error && (
        <div className="p-3 border-2 border-destructive rounded-sm bg-destructive/10 text-sm text-destructive">
          {error}
        </div>
      )}

      {/* Audio player */}
      {audioJobId && status === "completed" && (
        <div className="space-y-3 p-4 border-2 border-border rounded-sm bg-card">
          <audio controls className="w-full" src={getTtsAudioUrl(audioJobId)} />
          <div className="flex gap-2">
            <Button variant="outline" size="sm" asChild>
              <a href={getTtsAudioUrl(audioJobId)} download={`tts_${audioJobId}.wav`}>
                <Download className="h-4 w-4" />
                Download WAV
              </a>
            </Button>
          </div>
        </div>
      )}
    </div>
  )
}
```
